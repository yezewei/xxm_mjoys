# 智能质检技术方案

## 一、方案概述

### 1.1 功能定位
智能质检系统是位于**质检工作台**菜单下的核心功能模块，主要用于对客服通话录音进行自动化和人工质量检查，确保服务合规性和质量标准的执行。

### 1.2 核心能力
- **AI 智能质检**：基于大语言模型自动识别违规话术、服务问题
- **关键词质检**：基于规则引擎匹配预设关键词触发质检项
- **人工质检**：支持质检员手动标注和审核
- **多维度分析**：支持按模型、规则、任务等多维度统计和分析

### 4.2 关键词匹配引擎

```typescript
// 关键词匹配算法
function matchKeywords(content: string, rule: QualityRule): TriggeredRule | null {
  const { keywords } = rule
  const matchedKeywords: string[] = []
  
  // 去重并按长度排序（长词优先）
  const sortedKeywords = Array.from(new Set(keywords))
    .sort((a, b) => b.length - a.length)
  
  // 构建正则表达式
  const escapedKeywords = sortedKeywords.map(kw => 
    kw.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')
  )
  const regex = new RegExp(`(${escapedKeywords.join('|')})`, 'gi')
  
  const matches = content.match(regex)
  if (matches && matches.length > 0) {
    matchedKeywords.push(...Array.from(new Set(matches.map(m => m.toLowerCase()))))
    
    return {
      ruleCode: rule.code,
      ruleName: rule.name,
      ruleType: 'keyword',
      severity: rule.severity,
      matchedKeywords
    }
  }
  
  return null
}
```

### 4.3 对话高亮实现

```typescript
// 多规则关键词高亮
function highlightText(text: string, triggerRules: TriggeredRule[]): string {
  if (!triggerRules || triggerRules.length === 0) return text
  
  // 收集所有关键词
  const allKeywords = triggerRules.flatMap(rule => 
    rule.matchedKeywords || []
  )
  
  // 去重并按长度排序
  const uniqueKeywords = Array.from(new Set(allKeywords))
    .sort((a, b) => b.length - a.length)
  
  if (uniqueKeywords.length === 0) return text
  
  // 构建正则并替换
  const escapedKeywords = uniqueKeywords.map(kw => 
    kw.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')
  )
  const regex = new RegExp(`(${escapedKeywords.join('|')})`, 'gi')
  
  return text.replace(
    regex, 
    '<span class="highlight-keyword">$1</span>'
  )
}
```

```typescript
// AudioPlayer 组件核心逻辑
const useAudioPlayer = () => {
  const audioRef = ref<HTMLAudioElement | null>(null)
  const isPlaying = ref(false)
  const currentTime = ref(0)
  const duration = ref(0)
  const volume = ref(1)
  
  const loadAudio = (url: string) => {
    audioRef.value = new Audio(url)
    audioRef.value.addEventListener('timeupdate', () => {
      currentTime.value = audioRef.value!.currentTime
    })
    audioRef.value.addEventListener('loadedmetadata', () => {
      duration.value = audioRef.value!.duration
    })
  }
  
  const togglePlay = () => {
    if (!audioRef.value) return
    if (isPlaying.value) {
      audioRef.value.pause()
    } else {
      audioRef.value.play()
    }
    isPlaying.value = !isPlaying.value
  }
  
  const seek = (time: number) => {
    if (audioRef.value) {
      audioRef.value.currentTime = time
    }
  }
  
  return {
    loadAudio,
    togglePlay,
    seek,
    isPlaying,
    currentTime,
    duration,
    volume
  }
}
```

```sql
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_model (model_id),
    INDEX idx_code (code)
);

-- 质检任务表
CREATE TABLE quality_task (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255) NOT NULL COMMENT '任务名称',
    description TEXT COMMENT '任务说明',
    model_id BIGINT NOT NULL COMMENT '质检模型 ID',
    model_type ENUM('AI', 'MANUAL') NOT NULL COMMENT '质检类型',
    auditor_id BIGINT COMMENT '质检员 ID',
    target_object VARCHAR(255) COMMENT '质检对象',
    auto_append BOOLEAN DEFAULT FALSE COMMENT '是否自动追加',
    status ENUM('pending', 'running', 'completed', 'paused') DEFAULT 'pending',
    total_count INT DEFAULT 0 COMMENT '总任务数',
    ai_count INT DEFAULT 0 COMMENT 'AI 完成数',
    manual_count INT DEFAULT 0 COMMENT '人工完成数',
    filter_conditions JSON COMMENT '筛选条件',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_status (status),
    INDEX idx_model (model_id)
);

-- 质检任务明细表
CREATE TABLE quality_task_detail (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    task_id BIGINT NOT NULL COMMENT '任务 ID',
    call_id BIGINT NOT NULL COMMENT '通话 ID',
    ai_status ENUM('pending', 'completed', 'failed') DEFAULT 'pending',
    ai_result JSON COMMENT 'AI 质检结果',
    ai_summary TEXT COMMENT 'AI 质检总结',
    ai_retry_count INT DEFAULT 0 COMMENT 'AI 重试次数',
    manual_status ENUM('pending', 'completed') DEFAULT 'pending',
    manual_result JSON COMMENT '人工质检结果',
    manual_summary TEXT COMMENT '人工质检总结',
    is_modified BOOLEAN DEFAULT FALSE COMMENT '是否人工修改',
    trigger_rules JSON COMMENT '触发规则列表',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_task (task_id),
    INDEX idx_call (call_id),
    INDEX idx_ai_status (ai_status),
    INDEX idx_manual_status (manual_status)
);
```
